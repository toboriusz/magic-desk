    const url = new URL("http://localhost/api/auth/login");

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
}

    "email": "SbXb4PeCQBn2nZg2",
    "password": "TNb2XY7UYm0cAmhu",
    "remember_me": true
}
